% Integrate the equations of motion of a simple pendulum using Cartesian
% coordinates
%
% Inputs:
% tspan (nx1) Time span to integrate over. See ode45 documentation for
% details. Units of seconds.
% x0 (1x1) Initial x value. Units of meters.
% dx0 (1x1) Initial \dot{x} value. Units of meters/second.
% y0 (1x1) Initial y value. Units of meters.
% dy0 (1x1) Initial \dot{y} value. Units of meters/second.
%
% Outputs:
% t (nx1) Time output for numerical integration. See ode45 documentation
% for details.
% xs (nx1) Integrated time history for x coordinate (this is a column of
% the state history output of ode45 - z). Units of meters.
% dxs (nx1) Integrated time history for time derivative of the x
% coordinate (this is a column of z). Units of meters/second.
% ys (nx1) Integrated time history for y coordinate (this is a column of
% z). Units of meters.
% dys (nx1) Integrated time history for time derivative of the y
% coordinate (this is a column of z). Units of meters/second.
% theta (nx1) Time history of theta values computed from the values in z.
% Units of radians.
% thetadot (nx1) Time history of \dot\theta values computed from the
% values in z. Units of radians/second.
% ell (nx1) Time history of pendulum link length, computed from the
% values in z. Units of meters.
% gravity
g = 9.81; %m/s^2
%equations of motion in cartesian coords
function dz = simple_pendulum_cartesian_eom(t,z)
% your code goes here
% don't forget to keep your state definition consistent
% between this method, your initial conditions and your outputs
x_dot = z(2);
x_double_dot = (-z(1)*z(2)^2 +g*z(3)^2 -z(1)*z(4)^2 )/( (z(1)^2)+z(3)^2 );
y_dot = z(4);
y_double_dot = (-z(1)*g*z(3)^2 -(z(2)^2)*z(3) -z(3)*z(4)^2 )/( (z(1)^2)+z(3)^2 );
theta_dot = z(6);
theta_double_dot = -g*sin(z(5));
dz = [x_dot;x_double_dot;y_dot;y_double_dot;theta_dot;theta_double_dot];
end
% assemble initial state value
% this should be a column array
z0= [x0;dx0;y0;dy0;pi/6;0];
% integrate:
[t,z] = ode45(@simple_pendulum_cartesian_eom,tspan,z0);
% compute the outputs (all of these must be column arrays)
xs =z(1,:) ; % values of x throughout the integration
ys = z(3,:) ; % values of y throughout the integration
dxs = z(2,:) ; % values of \dot{x} throughout the integration
dys = z(4,:) ; % values of \dot{y} throughout the integration
ell = (xs.^2 + ys.^2).^0.5 ; % values of pendulum length throughout the integration, computed from ouptut state history z
theta = z(5,:) ; % values of pendulum angle throughout the integration, computed from ouptut state history z
thetadot = z(6,:) ; % values of \dot\theta throughout the integration
end
